Render

The simplest Z.render() will just fetch an HTML component and insert it into the "inner" element. You can also use "outer" to replace that element.

Z.render({
	html: "path/to/component.html",
	inner: document.getElementById("insert-here")
})

You can also add some data to the render, in the following example the array ["bar", "baz", "buff"] will be available globally as Z.foo

Z.render({
	html: "path/to/component.html",
	inner: document.getElementById("insert-here"),
	foo: ["bar", "baz", "buff"]    // optional
	foo: "path/to/data.json"	   // or as an external resource
})

Update

To update a component, with new data, simply do the following. This will re-render the component with the new Z.foo array

Z.update({
	foo: ["lorem", "ipsum", "dolor"]
})

You can also re-render with a new HTML component, if you so wish

Z.update({
	html: "path/to/new-component.html",
	foo: ["lorem", "ipsum", "dolor"]
})

Counter Demo

view live version or edit on glitch
<!-- index.html -->
<head>
    <script src="js/Z.min.js"></script>
</head>
<body>
	<h1>Counter Demo</h1>
    <button onclick="Z.update({count: Z.count - 1})">-</button>
    <span id="output"></span>
    <button onclick="Z.update({count: Z.count + 1})">+</button>
    <script>
        Z.render({
            html: "components/counter-total.html",
            inner: document.getElementById("output"),
            count: 0
        })
    </script>
</body>

<!-- counter-total.html -->
<script id="total">
    document.getElementById("total").outerHTML = Z.count;
</script>

Todo List Demo

view live version or edit on glitch
<!-- index.html -->
<head>
    <script src="js/Z.min.js"></script>
</head>
<body>
    <h1>Todo List Demo</h1>
    <div id="todo-list"></div>
    <div id="todo-add"></div>
    <script>
        Z.render({
            html: "components/todo-list.html",
            inner: document.getElementById("todo-list"),
            todos: []
        })

        Z.render({
            html: "components/todo-add.html",
            outer: document.getElementById("todo-add")
        })
    </script>
</body>

<!-- todo-list.html -->
<script id="list-container">
    document.getElementById("list-container").outerHTML = Z.todos.map(function(todo, index){
        return "<div><span>" + todo + "</span><button onclick=removeItem(" + index + ");>remove</button></div>";
    }).join("");

    function removeItem(index) {
        Z.todos.splice(index, 1);
        Z.update({
            todos: Z.todos
        })
    }
</script>

<!-- todo-add.html -->
<input id="task" type="text" placeholder="add task">
<button onclick="addItem()">Add</button>
<script>
	function addItem() {
		var task = document.getElementById("task");
		Z.todos.push(task.value);
		Z.update({
			todos: Z.todos
		})
		task.value = "";
	}
</script>
