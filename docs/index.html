The Simplest Component

This simple component will mount the returned html of the render function, replacing the DOM element of the "outer", in this example, the script element itself.

HTML

<script id="welcome">
  Z.mount({
    render: function() {
      return "<h1>Hello world!</h1>";
    },
    outer: document.getElementById("welcome")
  });
</script>
RESULT

Component with State

You can also add "state" to be used in the render of a component.

The state will be available globally after the component is mounted using the id. eg. Z.greeting will be {name: "Paul"}

HTML

<div id="welcome"></div>
<script>
  Z.mount({
    id: "greeting",
    render: function(state) {
      return `<h1>Hello ${state.name}!</h1>`
    },
    inner: document.getElementById("welcome"),
    state: {
      name: "Paul"
    }
  });
</script>
RESULT

Updating Components

Components can be updated by passing the id, This will then re-render that component.

HTML

<div id="welcome"></div>
<button onclick='Z.foo.update("Laura", 32)'>Update!</button>
<script>
  Z.mount({
    id: "foo",
    render: function(state) {
      return `<h1>${state.name} is ${state.age} years old!</h1>`;
    },
    inner: document.getElementById("welcome"),
    state: {
      name: "Robert",
      age: 48,
      update: function(name, age){
        this.name = name;
        this.age = age;
        Z.update("foo");
      }
    }
  });
</script>
RESULT

Loading External Data

You can also load data eg. a JSON before mounting a component.

The Z.load function takes an array of objects as the resources to be loaded, and a callback function fired when all the resources are fetched.

Resources are available globally as, in this example Z.load.robert

HTML

<div id="welcome"></div>
<script>
  Z.load(
    [
      {
        id: "robert",
        url: "json/robert.json",
        callback: function(data) {
          return JSON.parse(data);
        }
      }
    ],
    function() {
      Z.mount({
        id: "foo",
        render: function(state) {
          return `<h1>${state.name} is ${state.age} years old!</h1>`;
        },
        inner: document.getElementById("welcome"),
        state: Z.load.robert
      });
    }
  ) 
</script>
RESULT

Loading External Data and Caching

If you want to store data eg. using localStorage, there is a way to first query if there is local data before making a request

If the local method returns falsy, the request will be made, otherwise the returned data from the localStorage will be passed to the callback function.

HTML

<div id="welcome"></div>
<script>
  Z.load(
    [
      {
        id: "robert",
        url: "json/robert.json",
        local: function(){
          return localStorage.getItem("robert");
        },
        callback: function(data) {
          localStorage.setItem("robert", data)
          return JSON.parse(data);
        }
      }
    ],
    function() {
      Z.mount({
        id: "foo",
        render: function(state) {
          return `<h1>${state.name} is ${state.age} years old!</h1>`;
        },
        inner: document.getElementById("welcome"),
        state: Z.load.robert
      });
    }
  ) 
</script>
RESULT

Watching Components

It is also possible to watch for when components are updated.

NOTE: Z.watch must be declared after the component with that id has been mounted.

HTML

<div id="welcome"></div>
<button onclick='Z.foo.update("Luke", 30)'>Update!</button>
<script>
  Z.mount({
    id: "foo",
    render: function(state) {
      return `<h1>${state.name} is ${state.age} years old!</h1>`;
    },
    inner: document.getElementById("welcome"),
    state: {
      name: "Paul", 
      age: 35,
      update: function(name, age){
        this.name = name;
        this.age = age;
        Z.update("foo");
      }
    }
  });


  Z.watch("foo", function(state){
    alert(state.name + " use the force!");
  });
</script>
RESULT

Counter Demo

HTML

<h1>Counter Demo</h1>
<button onclick="Z.count.update(-1)">
  -
</button>
<span id="output"></span>
<button onclick="Z.count.update(1)">
  +
</button>
<script>
  Z.mount({
    id: "count",
    render: function(state) {
      return state.amount;
    },
    inner: document.getElementById("output"),
    state: {
      amount: 0,
      update: function(inc){
        this.amount += inc;
        Z.update("count");
      }
    }
  });
</script>

Todo Demo

HTML

<h1>Todo List Demo</h1>
<div id="todo-add"></div>
<div id="todo-list"></div>
<script>
  Z.mount({
    id: "list",
    render: function(state) {
      return state.todos
        .map(function(todo, index) {
          return `
            <div>
              <span>${todo}</span>
              <button onclick="Z.list.remove(${index})">remove</button>
            </div>
          `;
        })
        .join("");
    },
    inner: document.getElementById("todo-list"),
    state: {
      todos: [],
      add: function(){
        var task = document.getElementById("task");
        this.todos.push(task.value);
        task.value = "";
        Z.update("list");
      },
      remove: function(index){
        this.todos.splice(index, 1);
        Z.update("list");
      }
    }
  });

  Z.mount({
    render: function() {
      return `
        <input id="task" type="text" placeholder="add task" />
        <button onclick=Z.list.add()>Add</button>
      `;
    },
    outer: document.getElementById("todo-add")
  });
</script>
