ZED.JS

1kb micro frontend framework

The Simplest Component

This simple component will mount the returned html of the render function, replacing the DOM element of the "outer", in this example, the script element itself.

HTML

<script id="welcome">
  Z.mount({
    render: function() {
      return "<h1>Hello world!</h1>";
    },
    outer: document.getElementById("welcome")
  });
</script>
RESULT

Component with State

You can also add "state" to be used in the render of a component.

The state will be available globally after the component is mounted using the key. eg. Z.greeting will be {name: "Paul"}

HTML

<div id="welcome"></div>
<script>
  Z.mount({
    key: "greeting",
    render: function(state) {
      return `<h1>Hello ${state.name}!</h1>`
    },
    inner: document.getElementById("welcome"),
    state: {
      name: "Paul"
    }
  });
</script>
RESULT

Updating Components

Components can be updated by passing the key name, and an object with what is to be updated. This will then re-render that component.

HTML

<div id="welcome"></div>
<button onclick='Z.update("foo", {state: {name:"Tina", age:24} })'>Update!</button>
<script>
  Z.mount({
    key: "foo",
    render: function(state) {
      return `<h1>${state.name} is ${state.age} years old!</h1>`;
    },
    inner: document.getElementById("welcome"),
    state: {
      name: "Robert",
      age: 48
    }
  });
</script>
RESULT

Updating Components cont.

You are not just limited to updating state. It is also possible to update the render function as well.

NOTE: Only components mounted using "inner" can be updated, since mounting using "outer" will replace the original DOM element used for mounting.

HTML

<div id="welcome"></div>
<button id="update">Update!</button>
<script>
  Z.mount({
    key: "foo",
    render: function(state) {
      return `<h1>${state.name} is ${state.age} years old!</h1>`;
    },
    inner: document.getElementById("welcome"),
    state: {
      name: "Thomas", 
      age: 19
    }
  });
  document.getElementById("update").addEventListener("click", function(){
    Z.update("foo", {
      render: function(state){
        return `<h1>${state.age} years old ${state.name} is!</h1>`;
      },
      state: {
        name: "Luke", 
        age: 31
      }
    });
  })
</script>
RESULT

Loading External Data

You can also load data eg. a JSON before mounting a component.

The Z.load function takes an array of objects as the resources to be loaded, and a callback function fired when all the resources are fetched.

Resources are available globally as, in this example Z.load.robert

HTML

<div id="welcome"></div>
<script>
  Z.load(
    [
      {
        key: "robert",
        url: "demos/json/robert.json",
        callback: function(data) {
          return JSON.parse(data);
        }
      }
    ],
    function() {
      Z.mount({
        key: "foo",
        render: function(state) {
          return `<h1>${state.name} is ${state.age} years old!</h1>`;
        },
        inner: document.getElementById("welcome"),
        state: Z.load.robert
      });
    }
  ) 
</script>
RESULT

Watching Components

It is also possible to watch for when components are updated.

NOTE: Z.watch must be declared after the component with that key has been mounted.

HTML

<div id="welcome"></div>
<button onclick='Z.update("foo", {state: {name:"Luke", age:30} })'>Update!</button>
<script>
  Z.mount({
    key: "foo",
    render: function(state) {
      return `<h1>${state.name} is ${state.age} years old!</h1>`;
    },
    inner: document.getElementById("welcome"),
    state: {name: "Paul", age: 35}
  });

  Z.watch("foo", function(){
    alert(Z.foo.name + " use the force!");
  });
</script>
RESULT

Before and After

The functions before and after are fired before and after the render of a component.

If you return false from the before function the component will not render.

This can be used to only re-render a component if the state is different from the previous state.

HTML

<div id="welcome"></div>
<script>
  Z.mount({
    key: "foo",
    before: function(state, prev){
      return JSON.stringify(state) != JSON.stringify(prev);
    },
    render: function(state, prev) {
      console.log("render fired");
      return `
        <h1>${state.name} is ${state.age} years old!</h1>
        <button id="btn">Update!</button>
      `;
    },
    after: function(){
      document.getElementById("btn").addEventListener("click", function(){
        Z.update("foo", {state: {
          name: "Thomas",
          age: 72
        }})
      })
    },
    inner: document.getElementById("welcome"),
    state: {
      name: "Benita",
      age: 36
    }
  });  
</script>
RESULT

HTML

<h1>Todo List Demo</h1>
<div id="todo-add"></div>
<div id="todo-list"></div>
<script>
  Z.mount({
    key: "todos",
    render: function(state) {
      return state
        .map(function(todo, index) {
          return `
            <div>
              <span>${todo}</span>
              <button onclick="Z.todos.splice(${index}, 1); Z.update('todos', {state: Z.todos})">remove</button>
            </div>
          `;
        })
        .join("");
    },
    inner: document.getElementById("todo-list"),
    state: []
  });

  Z.mount({
  render: function() {
    return `
      <input id="task" type="text" placeholder="add task" />
      <button>Add</button>
    `;
  },
  after: function() {
    document
      .querySelector("#task + button")
      .addEventListener("click", function() {
        var task = document.getElementById("task");
        Z.todos.push(task.value);
        Z.update("todos", {
          state: Z.todos
        });
        task.value = "";
      });
  },
  outer: document.getElementById("todo-add")
  });
</script>
RESULT
